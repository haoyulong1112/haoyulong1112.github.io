    # 计算属性

    var vm = new
Vue({
 el: '#example',
 data: {
   message: 'Hello'
 },
  computed: {
    // 计算属性的 getter
    reversedMessage: function () {
      // `this` 指向 vm 实例
      return this.message.split('').reverse().join('')
    }
  }
})

这里我们声明了一个人计算属性reversedMessage。我们提供的函数将用作属性vm.reversedMessage的getter函数：
    vm.message = 'Goodbye'
    console.log(vm.reversedMessage)
